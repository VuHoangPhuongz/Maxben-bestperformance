    <script>
        document.addEventListener('DOMContentLoaded', () => {
            // --- Active Nav Link on Scroll ---
            // This script highlights the navigation link that corresponds to the section currently visible on screen.
            
            const sections = document.querySelectorAll('section[id]');
            const navLinks = document.querySelectorAll('header nav a');
            
            if (!sections.length || !navLinks.length) return;

            const headerHeight = document.querySelector('header').offsetHeight;

            // Options for the Intersection Observer.
            // rootMargin determines how far from the viewport edge the intersection is calculated.
            // Here, it creates a horizontal "detection line" near the top of the viewport, offset by the header height.
            const observerOptions = {
                root: null,
                rootMargin: `-${headerHeight}px 0px -50% 0px`,
            };

            // The observer callback function.
            const sectionObserver = new IntersectionObserver((entries, observer) => {
                entries.forEach(entry => {
                    // When a section enters the detection area, it's considered "intersecting".
                    if (entry.isIntersecting) {
                        const link = document.querySelector(`header nav a[href="#${entry.target.id}"]`);
                        
                        // Remove 'active' from all links.
                        navLinks.forEach(l => l.classList.remove('active'));
                        
                        // Add 'active' to the link corresponding to the visible section.
                        if (link) {
                           link.classList.add('active');
                        }
                    }
                });
            }, observerOptions);

            // Observe all sections with an ID.
            sections.forEach(section => {
                sectionObserver.observe(section);
            });

            // --- Clear active link when at the top of the page ---
            const heroSection = document.querySelector('.hero-bg');
            const heroObserver = new IntersectionObserver((entries) => {
                // If more than 20% of the hero section is visible, remove active state from all nav links.
                if (entries[0].isIntersecting) {
                    navLinks.forEach(link => link.classList.remove('active'));
                }
            }, { threshold: 0.2 }); 
            
            if (heroSection) {
                heroObserver.observe(heroSection);
            }
        });
    </script>
